<script>
  const questions = [
    { q: "Which animal is known as the king of the jungle?", options: ["Elephant", "Lion", "Tiger", "Bear"], answer: 1 },
    { q: "What is the largest mammal?", options: ["Elephant", "Blue Whale", "Giraffe", "Hippo"], answer: 1 },
    { q: "Which bird is a symbol of peace?", options: ["Crow", "Dove", "Eagle", "Parrot"], answer: 1 }
  ];

  let coins = 0;

  function showScreen(id) {
    document.querySelectorAll('.screen').forEach(s => s.classList.remove('active'));
    document.getElementById(id).classList.add('active');
  }

  function startQuiz() {
    showScreen('quiz');
    const qDiv = document.getElementById('questions');
    qDiv.innerHTML = "";
    questions.forEach((q, i) => {
      let html = `<p>${q.q}</p>`;
      q.options.forEach((opt, j) => {
        html += `<label><input type="radio" name="q${i}" value="${j}"> ${opt}</label><br>`;
      });
      qDiv.innerHTML += html;
    });
  }

  function submitQuiz() {
    let correct = 0;
    questions.forEach((q, i) => {
      const selected = document.querySelector(`input[name="q${i}"]:checked`);
      if (selected && parseInt(selected.value) === q.answer) {
        correct++;
      }
    });
    const earned = correct * 20; // 1 correct = 20 coins (max 60)
    coins += earned;
    document.getElementById('score').innerText = `You got ${correct} out of ${questions.length} correct!`;
    document.getElementById('coins').innerText = `You earned ${earned} coins.`;
    showScreen('results');
  }

  function openShop() {
    document.getElementById('coinBalance').innerText = coins;
    showScreen('shop');
  }

  function buyItem(cost) {
    if (coins >= cost) {
      coins -= cost;
      alert("Purchase successful!");
      document.getElementById('coinBalance').innerText = coins;
    } else {
      alert("Not enough coins!");
    }
  }

  function restart() {
    coins = 0;
    showScreen('menu');
  }
</script>
